int **warschall_algorithm(int n, int R[n][n])
{
  int T[n][n];
  int **S = R;
  for(int i = 0; i < n; ++i)
  {
    for(int j = 0; j < n; ++j)
    {
      for(int k = 0; k < n; ++k)
        T[j][k] = S[j][k] | (S[j][i] & S[i][k]);
    }
    S = T;
  }
  return T;
}
