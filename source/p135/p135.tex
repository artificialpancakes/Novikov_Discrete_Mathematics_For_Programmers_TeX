  for (int i = 0; i < pow(2, n2); i++) {

      } else {
        f = true; /* это первое слагаемое в дизъюнкции~*/
      }
    }
    bool g = false; /* признак присутствия левого операнда конъюнкции~*/

      if (g) {

        g = true; /* это первый сомножитель в конъюнкции~*/
